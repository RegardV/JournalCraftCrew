### **Real-time Data Flow**
```javascript
// WebSocket connection for live updates
const orchestratorSocket = new WebSocket('ws://localhost:6770/ws/orchestrator');

// 10-second update interval
setInterval(() => {
    orchestratorSocket.send(JSON.stringify({
        type: 'get_dashboard_update',
        timestamp: Date.now()
    }));
}, 10000);

// Handle real-time updates
orchestratorSocket.onmessage = (event) => {
    const data = JSON.parse(event.data);
    updateDashboard(data);

    // Animate status changes
    if (data.agent_status_changed) {
        animateAgentStatus(data.agent, data.new_status);
    }

    // Flash alerts for critical issues
    if (data.critical_alert) {
        showCriticalAlert(data.alert);
    }
};
```
